for i in range(int(input())):
    s = list(map(int,list(input())))
    n = len(s)
    p = str(s[0])
    r = ""
    x = 0
    if s[1] == 0:
        for i in range(1,n):
            if s[i] != 0:
                x = i
                break
            while s[i] == 0 and i+1 < n:
                i += 1
                if i+1 < n and s[i+1] != 0:
                    x = i+1
                    break
            p += str(s[i])
        for i in range(x,n):
            r += str(s[i])

        if int(p) < int(r):
            print(p,r)
        else:
            print(-1)
    else:
        a = s[0]
        b = ""
        for i in range(1,n):
            b += str(s[i])
        print(a,b)
