for i in range(int(input())):
    s = list(map(int,list(input())))
    n = len(s)
    p = str(s[0])
    r = ""
    x = 0
    if s[1] == 0:
        for i in range(1,n):
            if s[i] != 0:
                x = i
                break
            else:
                p += str(s[i])
        while x<n:
            if s[x] == 0:
                x += 1
                continue
            r += str(s[x])
            x += 1

        if int(p) < int(r):
            print(p,r)
        else:
            print(-1)
    else:
        a = s[0]
        b = ""
        for i in range(1,n):
            b += str(s[i])
        print(a,b)
