for i in range(int(input())):
    s = list(map(int,list(input())))
    n = len(s)
    p = str(s[0])
    r = ""
    x = 0
    if s[1] == 0:
        for i in range(1,n):
            if s[i] != 0:
                x = i
                break
            else:
                p += str(s[i])
        for i in range(x,n):
            r += str(s[i])

        while int(p)<int(r):
            i = len(p)
            p = p[:i-1]+str(int(p[i-1])+1)
            p = p+'0'*(len(r)-len(p))

            print(p,r)
        else:
            print(-1)
    else:
        a = s[0]
        b = ""
        for i in range(1,n):
            b += str(s[i])
        print(a,b)
