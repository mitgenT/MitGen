def flip_case_special(string: str, special_index: int) -> str:
    special_index %= len(string)
    result = ''
    for i, char in enumerate(string):
        if i == special_index:
            if char.isupper():
                result += char.lower()
            else:
                result += char
        else:
            if char.isupper():
                # Flip to lowercase if not at the special index
                result += char.lower() if i != special_index else char
            else:
                result += char
    return result
