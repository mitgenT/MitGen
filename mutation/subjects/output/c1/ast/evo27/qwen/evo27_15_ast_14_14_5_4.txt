def flip_case_special(string: str, special_index: int) -> str:
    special_index %= len(string)
    result = ''
    for i, char in enumerate(string):
        if i == special_index:
            if char.isupper():
                result += char.lower()
            else:
                result += char
        else:
            if char.isupper():
                result += char.lower()
            else:
                # Keep lowercase characters unchanged at the 'special_index' position
                result += char if i != special_index else char.upper()
    return result
