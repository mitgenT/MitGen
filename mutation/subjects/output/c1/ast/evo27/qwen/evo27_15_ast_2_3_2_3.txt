def flip_case_special(string: str, special_index: int) -> str:
    result = ""
    for i, char in enumerate(string):
        if i == special_index:
            if char.isupper():
                result += char.lower()
            else:
                result += char
        else:
            if char.isalpha():
                result += char.swapcase()
            else:
                result += char
    return result
