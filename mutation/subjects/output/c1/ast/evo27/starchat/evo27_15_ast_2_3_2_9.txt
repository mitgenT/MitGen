def flip_case_special(string: str, special_index: int) -> str:
    result = ""
    len_string = len(string)
    special_index = special_index % len_string
    for i, char in enumerate(string):
        if i == special_index:
            if char.isupper():
                result += char.lower()
            else:
                result += char
        else:
            if char.isupper():
                result += char.lower()
            else:
                result += char.upper()
    for i, char in enumerate(string):
        if i == special_index:
            if char.isupper():
                result += char.lower()
            else:
                result += char
        else:
            if char.isupper():
                result += char.lower()
            else:
                result += char
    return result
