t=int(input())
for i in range(t):
    n=int(input())
    a=[int(ele) for ele in input().split()]
    b=[int(ele) for ele in input().split()]
    d=[[a[i]+b[i],i] for i in range(n)]
    res = sorted(d, key=lambda x: x[0], reverse=True)
    res=0
    for i in range(n):
        if i%2:res-=(b[d[i][1]]-1)
        else:res+=(b[d[i][1]]-1)
    print(res)
