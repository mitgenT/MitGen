t=int(input())
for i in range(t):
    n=int(input())
    a=list(map(int,input().split()))
    b=list(map(int,input().split()))
    d=[(a[i]-b[i],i) for i in range(n)]
    d=sorted(d)
    res=0
    for i in range(n):
        if i%2:res-=(b[d[i][1]]-1)
        else:res+=(b[d[i][1]]-1)
    print(res)
