def string_sequence_modified(n: int, m: int) -> str:
    result = []
    for i in range(n):
        if (i+1) % m == 0:
            result.append("Fizz")
        else:
            result.append(str(i))
    if result:  # Check if the list is not empty
        result = " ".join(result)  # Concatenate elements with spaces
