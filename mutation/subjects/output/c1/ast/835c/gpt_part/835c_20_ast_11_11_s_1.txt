
t = int(input())
for _ in range(t):
    n = int(input())
    s = [int(x) for x in input().split()]
    mx = [0,0]
    for i in range(n):
        if s[i]>=s[mx[0]]:
            mx[1] = mx[0]
            mx[0] = i
        if i==mx[1]:
            mx[1] = mx[0]
            mx[0] = i
            mx[1] = i
    for i in range(n):
        if i==mx[0]:
            print(s[i]-s[mx[1]],end=" ")
        else:
            print(s[i]-s[mx[0]],end=" ")
    print()
