
for _ in range (int(input())):
    n,c,d=map(int,input().split())
    A=list(map(int,input().split()))
    A=sorted(A)[::-1]
    if sum(A[:d])>=c:
        print("Infinity")
    elif A[0]*d<c:
        print("Impossible")
    else:
        l,r=0,d
        mid=(l+r+1)//2
        while l<r:
            selecsum=sum(A[:mid+1])
            numselec=d//(mid+1)
            remain=d%(mid+1)
            coins=numselec*selecsum+sum(A[:remain])
            if coins>=c:
                l=mid
            elif coins < c:
                r = mid - 1
                r=mid-1
            mid=(l+r+1)//2
        print(mid-1)
