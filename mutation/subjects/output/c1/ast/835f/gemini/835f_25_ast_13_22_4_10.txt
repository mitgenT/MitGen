
for _ in range (int(input())):
    n,c,d=map(int,input().split())
    A=list(map(int,input().split()))
    A=sorted(A)[::-1]
    if sum(A[:d])>=c:
        print("Infinity")
    elif A[0]*d<c:
        print("Impossible")
    else:
        l,r=0,d
        mid=(l+r+1)//2
        if sum(A[:mid]) >= c:
            l = mid
        else:
            r = mid
            mid = (l + r + 1) // 2
        print(mid-1)
