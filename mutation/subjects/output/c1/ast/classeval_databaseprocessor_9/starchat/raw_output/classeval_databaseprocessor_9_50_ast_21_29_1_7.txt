        conn.commit()

        select_query = f"SELECT * FROM {table_name} WHERE name =?"
        cursor.execute(select_query, (name,))
        result = cursor.fetchall()

        return result

    def delete_from_database(self, table_name, name):

        conn = sqlite3.connect(self.database_name)
